#!/usr/local/bin/perl
#
# 注意: 実行する場合は、たぶんこのスクリプト自体と入力ファイルを *euc-japan*
#       に変換しておく必要がある(手抜き)。
#
#	$Id: terms-freebsd-to-netbsd.txt,v 1.1 1999/01/08 11:42:34 tsubai Exp $
#

undef($/);
$body = <>;
$/ = "\n";

while (<DATA>) {
	next if (/^#/ || /^$/);
	($eng, $orig, $mod) = split;

	$body =~ s/$orig\($modでなく\)/$orig/gm;	# 用語集専用
	$body =~ s/$orig/$mod/gm;
}

print $body;

__END__
